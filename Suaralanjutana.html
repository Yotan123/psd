def extract_features(file_path, sr=22050):
    y, sr = librosa.load(file_path, sr=sr)

    # durasi tidak lebih dari 1 detik
    max_len = int(sr * 1.0)
    if len(y) < max_len:
        y = np.pad(y, (0, max_len - len(y)), mode='constant')
    else:
        y = y[:max_len]

    y = y / (np.max(np.abs(y)) + 1e-6) # Normalisasi

    feat = {
        "mean": np.mean(y),
        "var": np.var(y),
        "skew": stats.skew(y),
        "kurt": stats.kurtosis(y),
        "rms": np.sqrt(np.mean(y**2)),
        "zcr": np.mean(librosa.feature.zero_crossing_rate(y)),
        "centroid": np.mean(librosa.feature.spectral_centroid(y=y, sr=sr)),
        "bandwidth": np.mean(librosa.feature.spectral_bandwidth(y=y, sr=sr)),
        "rolloff": np.mean(librosa.feature.spectral_rolloff(y=y, sr=sr))
    }

    hist, _ = np.histogram(y, bins=50, density=True)
    hist = hist[hist > 0]
    feat["entropy"] = -np.sum(hist * np.log2(hist + 1e-10))
    return feat

def find_audio_file_from_gabungan():
    # Memilih dari folder Gabungan_Suara
    buka_folder_gabungan = os.path.join(TARGET_DIR, 'buka')
    tutup_folder_gabungan = os.path.join(TARGET_DIR, 'tutup')

    audio_files_buka = []
    if os.path.exists(buka_folder_gabungan):
        audio_files_buka = [f for f in os.listdir(buka_folder_gabungan) if f.lower().endswith('.wav')]

    audio_files_tutup = []
    if os.path.exists(tutup_folder_gabungan):

# Fungsi untuk ekstraksi fitur (konsisten dengan sebelumnya)
def extract_features(file_path, sr=22050):
    y, sr = librosa.load(file_path, sr=sr)

    # durasi tidak lebih dari 1 detik
    max_len = int(sr * 1.0)
    if len(y) < max_len:
        y = np.pad(y, (0, max_len - len(y)), mode='constant')
    else:
        y = y[:max_len]

    y = y / (np.max(np.abs(y)) + 1e-6) # Normalisasi

    feat = {
        "mean": np.mean(y),
        "var": np.var(y),
        "skew": stats.skew(y),
        "kurt": stats.kurtosis(y),
        "rms": np.sqrt(np.mean(y**2)),
        "zcr": np.mean(librosa.feature.zero_crossing_rate(y)),
        "centroid": np.mean(librosa.feature.spectral_centroid(y=y, sr=sr)),
        "bandwidth": np.mean(librosa.feature.spectral_bandwidth(y=y, sr=sr)),
        "rolloff": np.mean(librosa.feature.spectral_rolloff(y=y, sr=sr))
    }

    hist, _ = np.histogram(y, bins=50, density=True)
    hist = hist[hist > 0]
    feat["entropy"] = -np.sum(hist * np.log2(hist + 1e-10))
    return feat
